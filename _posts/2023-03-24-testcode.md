void delay(int n) {
	volatile int i, j;
	for (i = 0; i < 100; i++) {
		for (j = 0; j < n; j++);
	}
}

volatile int upper(int n) {
	volatile int result = 0x0000;
	for (int j = 0; j < 9 - n; j++) {
		result = num[j] | result;
	}
	return result;

}

volatile int lower(int n) { // 앞에서 부터 coutning
	volatile int result = 0x0000;
	for (int j = 0; j < n; j++) {
		result = num[9-j] | result;
	}
	return result;

}

int main(void) {
	MCUCR = 0x80;
	DM_SEL = 0x0000;
	DM_DATA = 0x0000;
	
	while (1) {
		for (int t = 0; t < Time; t++) {
			for (int i = 0; i < 10; i++) {
				DM_SEL = num[i];
				DM_DATA = upper(i);
				delay(2);
			}
		}
		for (int t2 = 0; t2 < Time; t2++) {
			for (int k = 0; k < 10; k++) {
				DM_SEL = num[k];

void MoveHammer(volatile int result, volatile int* hammer, int* hammer_row, int* hammer_column){
	switch(result){
		case 0x01: {	// SW0 - Go right
			if( *(hammer_column) != 0 )
			{
				hammer[*(hammer_row)] = hammer[*(hammer_row)] >> 1;
				(*(hammer_column))--;
			}
			break;
		}
		case 0x02: {	// SW1 - Go left
			if( *(hammer_column) != 9 ){
				hammer[*(hammer_row)] = hammer[*(hammer_row)] << 1;
				(*(hammer_column))++;
			}
			break;
		}
		case 0x04: {	// SW2 - Go down
			if( *(hammer_row) != 9 ){
				hammer[*(hammer_row)+1]=hammer[*(hammer_row)];
				hammer[*(hammer_row)]=0x0000;
				(*(hammer_row))++;
			}
			break;
		}
		case 0x08: {	// SW3 - Go up
			if( *(hammer_row) != 0 ){
				hammer[*(hammer_row)-1]=hammer[*(hammer_row)];
				hammer[*(hammer_row)]=0x0000;
				(*(hammer_row))++;
			}
			break;
		}
	}
}
